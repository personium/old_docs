 Node type

 D:
 Element

 D:
 Element

 Element

Request Sample

<D:propertyupdate xmlns:D="DAV:"
    xmlns:p="urn:x-personium:xmlns"
    xmlns:Z="http://www.w3.com/standards/z39.50/">
    <D:set>
        <D:prop>
            <Z:Author>Author1 update</Z:Author>
            <p:hoge>fuga</p:hoge>
        </D:prop>
    </D:set>
    <D:remove>
        <D:prop>
            <Z:Author/>
            <p:hoge/>
        </D:prop>
    </D:remove>
</D:propertyupdate>
